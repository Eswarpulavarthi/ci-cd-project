pipeline {
    agent any

    stages {

        stage('Build') {
            steps {
                bat 'mvn clean package'
            }
        }

        stage('Docker Build') {
            steps {
                bat 'docker build -t ci-cd-app .'
            }
        }

        stage('Run Container') {
            steps {
                bat 'docker run -d -p 8090:8080 ci-cd-app'
            }
        }
    }
}
